with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as zf:
    for root, dirs, files in os.walk("praharsh_siem_project"):

            fpath = os.path.join(root, file)
            arcname = os.path.relpath(fpath, "praharsh_siem_project")
            zf.write(fpath, arcname)

print(f"📦 ZIP rebuilt: {os.path.getsize(zip_path)/1024/1024:.2f} MB")
